파이썬 내 re 패키지를 활용한다.

import re
re.findall("abc","kafeiaselfabcalisjeflajslefj;abc") # 파이썬의 리스트 형태로

iterator = re.finditer("abc","kafeiaselfabcalisjeflajslefj;abc")
# ['abc','abc']

for match in iterator:
  print(match.group())
# abc
# abc

# [ ] : 일치 시키려는 문자 집합인 문자 클래스를 지정하는데 사용
re.findall("abc","askdfj;akdfj;lajdf;laj;lakdsfabcasdfaabcdadf")
# ['abc', 'abc']
