(2) smtplib 패키지 설치 : 메일 서버 접속 패키지이다.
보내는 사람 주소는 gmail이어야 한다.

  • 파일 첨부해서 이메일 보내기 기본 코드
# 메일 서버 접속 패키지 smtplib
# 보내는 사람 이메일 주소를 gmail 사용해야 함

import smtplib
from email.mime.multipart import MIMEMultipart
from email.mime.base import MIMEBase
from email.mime.text import MIMEText
from email import encoders


# 보내는 사람 정보
me = "    @gmail.com" # gmail만
my_password = "          "

# 로그인하기
s = smtplib.SMTP_SSL('smtp.gmail.com')
s.login(me, my_password)

# 받는 사람 정보
you = "      @naver.com"  # 아무_도메인

# 메일 기본 정보 설정
msg = MIMEMultipart('alternative')
msg['Subject'] = "기사 리스트" # 제목
msg['From'] = me
msg['To'] = you

# 메일 내용 쓰기
content = "크롤링 결과물 입니다." # 메일 내용
part2 = MIMEText(content, 'plain')
msg.attach(part2)

# 파일 첨부 하기
# 여러 파일을 첨부할 때에는 똑같은 코드를 반복하여 각각 msg.attach()
part = MIMEBase('application', "octet-stream")
with open("   .xlsx", 'rb') as file:
    part.set_payload(file.read())
encoders.encode_base64(part)
part.add_header('Content-Disposition', "attachment", filename="   .xlsx") # 보낼 파일명
msg.attach(part)

# 메일 보내고 서버 끄기
s.sendmail(me, you, msg.as_string())

s.quit()

# 메일 서버 접속 패키지 smtplib
# 보내는 사람 이메일 주소를 gmail 사용해야 함

import smtplib
from email.mime.multipart import MIMEMultipart
from email.mime.base import MIMEBase
from email.mime.text import MIMEText
from email import encoders


# 보내는 사람 정보
me = "생략@gmail.com" (어쨌든 gmail)
my_password = "  " (비밀번호 입력)

# 로그인하기
s = smtplib.SMTP_SSL('smtp.gmail.com')
s.login(me, my_password)

# 받는 사람 정보
# 아무_도메인이든 상관없다
# 여러 사람에게 보내려면 이렇게 반복문
email_list = ["    @naver.com","   @naver.com"]

for you in email_list:
    # 메일 기본 정보 설정
    msg = MIMEMultipart('alternative')
    msg['Subject'] = "       " # 제목
    msg['From'] = me
    msg['To'] = you

    # 메일 내용 쓰기
    content = "           " # 메일 내용
    part2 = MIMEText(content, 'plain')
    msg.attach(part2)

    # 파일 첨부 하기
    # 여러 파일을 첨부할 때에는 똑같은 코드를 반복하여 각각 msg.attach()
    part = MIMEBase('application', "octet-stream")
    with open("    .xlsx", 'rb') as file: # 로컬에 있는 파일명
        part.set_payload(file.read())
    encoders.encode_base64(part)
    part.add_header('Content-Disposition', "attachment", filename="   .xlsx") # 실제로 보낼 파일명
    msg.attach(part)

    # 메일 보내고 서버 끄기
    s.sendmail(me, you, msg.as_string())

s.quit()

with open("text.txt","w",encoding+utf-8) as f: # r 읽기모드, w 쓰기모드
   f.write("각 줄마다 번호를 적은 파일입니다.\n") # 쓰기모드는 덮어씌우니 주의
   for i in [1,2,3,4,5]:
       f.write(f"이것은 {i}번째 줄입니다.\n")
f.close()

(2) 카톡 대화 데이터 전처리 작업 = 노가다

text = ''
with open("kakaotalk.txt", "r", encoding="utf-8") as f:
    lines = f.readlines()
    for line in lines[5:]: # 처음부터 안 읽고
        if '] [' in line: # 사람들 대화 특징만 보고 => ] [
            text += line.split('] ')[2].replace('저도','').replace('저두','').replace('저는','').replace('ㅋ','').replace('ㅠ','').replace('이모티콘\n','').replace('사진\n','')

print(text)
