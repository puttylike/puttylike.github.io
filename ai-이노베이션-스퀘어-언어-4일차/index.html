import spacy

text = "Let's go!"

# lemma, stem
# lemma : 단어의 base form을 찾는 것
# stem : 단어의 어근을 의미
# stem의 예 : university라는 단어의 lemma는 (university), stem(universe)
# stem의 예 : universe라는 단어의 lemma는 (universe), stem(univer)
# stem은 자체로는 완전한 단어의 형태는 아니지만, 의미는 가지고 있다.

# Lemma Algorithm : 통사적인 문법적인 접근을 통해서 분석하는 방식, dictionary 같은 vocabulary가 필요

# WSD : Word-sense disambiguation 단어 의미 중의성 해소
# 단어들이 가진 중의성으로 인해 사람들도 다른 의미로 해석을 한다.
# WSD problem : 자연어 처리에서 open problem, 복잡하고 통계적인 모델을 계속 연구하고 발표하고 있다.
#
# Bass - fish (Noun)
# Bass - lowest male voice (Noun)
# Bass - guitar (Noun)
#
# fish - Verb
# fish - Noun

# sentence1 = "I frew to Rome."
# sentence2 = "I'm flying to Rome."
# sentence3 = "I will fly to Rome."


sentence1 = "I frew to Rome."
sentence2 = "I'm flying to Rome."
sentence3 = "I will fly to Rome."

doc1 = nlp(sentence1)
doc2 = nlp(sentence2)
doc3 = nlp(sentence3)

for doc in [doc1, doc2, doc3]:
    print( [(w.text, w.lemma_) for w in doc if w.tag_ == "VBG" or w.tag_ == "VB"]) # 현재진행형, 현재형

# 실행 결과
[]
[('flying', 'fly')]
