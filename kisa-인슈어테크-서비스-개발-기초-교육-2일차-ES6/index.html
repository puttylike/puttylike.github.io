// object
var car01 = {
  // key : value
  name : "bmw",
  ph : 120,
  start : function(){
    console.log('engine start');
  }
};

car01.start(); // engine start
console.log(car01.ph); // 120
console.log(car01); // { name: 'bmw', ph: 120, start: [Function: start] }
